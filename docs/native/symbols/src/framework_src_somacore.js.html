 34 		Function.prototype.bind = function bind(that) {
 35 			var target = this;
 36 			if (typeof target != "function") {
 37 				throw new Error("Error, you must bind a function.");
 38 			}
 39 			var args = Array.prototype.slice.call(arguments, 1); // for normal call
 40 			var bound = function () {
 41 				if (this instanceof bound) {
 42 					var F = function(){};
 43 					F.prototype = target.prototype;
 44 					var self = new F;
 45 					var result = target.apply(
 46 						self,
 47 						args.concat(Array.prototype.slice.call(arguments))
 48 					);
 49 					if (Object(result) === result) {
 50 						return result;
 51 					}
 52 					return self;
 53 				} else {
 54 					return target.apply(
 55 						that,
 56 						args.concat(Array.prototype.slice.call(arguments))
 57 					);
 58 				}
 59 			};
 60 			return bound;
 61 		};
 62 	}

 70 	soma.inherit = function(target, obj) {
 71 		var subclass;
 72 		if (obj && obj.hasOwnProperty('constructor')) {
 73 			// use constructor if defined
 74 			subclass = obj.constructor;
 75 		} else {
 76 			// call the super constructor
 77 			subclass = function(){
 78 				return target.apply(this, arguments);
 79 			};
 80 		}
 81 		// add super properties
 82 		soma.applyProperties(subclass.prototype, target.prototype);
 83 		// set the prototype chain to inherit from the parent without calling parent's constructor
 84 		var chain = function(){};
 85 		chain.prototype = target.prototype;
 86 		subclass.prototype = new chain();
 87 		// add obj properties
 88 		if (obj) soma.applyProperties(subclass.prototype, obj, target.prototype);
 89 		// point constructor to the subclass
 90 		subclass.prototype.constructor = subclass;
 91 		// set super class reference
 92 		subclass.parent = target.prototype;
 93 		// add extend shortcut
 94 		subclass.extend = function(obj) {
 95 			return soma.inherit(subclass, obj);
 96 		};
 97 		return subclass;
 98 	};

192 	soma.AutoBind = {
193 		blackList: ["initialize", "parent", "constructor", "$constructor", "addEventListener", "removeEventListener"],
194 		autobind: function() {
195 			if (this.wasAutoBound) {
196 				return;
197 			}
198 			var o = this;
199 			var ab = o["AutoBindPattern"];
200 			var coreAb = "([lL]istener|[hH]andler|[cB]allback)$";
201 			if (!ab) {
202 				ab = coreAb;
203 			} else {
204 				ab = coreAb + "|" + ab;
205 			}
206 			for (var k in o) {
207 				if (typeof o[k] == "function") {
208 					if (this._autobindIsBlacklisted(k)) {
209 						continue;
210 					}
211 					if (!k.match(ab)) {
212 						continue;
213 					}
214 					o[k] = o[k].bind(o);
215 				}
216 			}
217 		},
218 		_autobindIsBlacklisted: function(name) {
219 			var bl = this.blackList;
220 			for (var i = 0; i < bl.length; i++) {
221 				if (bl[i] == name) {
222 					return true;
223 				}
224 			}
225 			return false;
226 		}
227 	};

251 		commands: null,
252 		currentCommand: null,
253 		id:null,
254 		constructor: function(id) {
255 			if (id == null) {
256 				throw new Error("SequenceCommand Children expect an unique id as constructor arg");
257 			}
258 			this.commands = [];
259 			this.id = id;
260 			soma.Command.call(this);
261 		},
262 		registerInstance: function(instance) {
263 			this.instance = instance;
264 			this.initializeSubCommands();
265 		},
266 		initializeSubCommands: function() {
267 			throw new Error("Subclasses of SequenceCommand must implement initializeSubCommands()");
268 		},
269 		addSubCommand: function(event) {
270 			var c = new SequenceCommandProxy(event);
271 			this.commands.push(c);
272 			this.instance.controller.registerSequencedCommand(this, c);
273 		},
274 		execute: function(event) {
275 			if (this.commands == null || this.commands.length === 0) {
276 				return;
277 			}
278 			this.currentCommand = this.commands.shift();
279 			if (this.hasCommand(this.currentCommand.event.type)) {
280 				this.dispatchEvent(this.currentCommand.event);
281 			}
282 		},
283 		executeNextCommand: function() {
284 			if (this.commands == null) {
285 				return;
286 			}
287 			this.instance.controller.unregisterSequencedCommand(this, this.currentCommand.event.type);
288 			if (this.commands.length > 0) {
289 				this.execute(this.commands[0].event);
290 			} else {
291 				this.commands = null;
292 				this.currentCommand = null;
293 			}
294 		},
295 		getLength: function() {
296 			if (this.commands == null) {
297 				return -1;
298 			}
299 			return this.commands.length;
300 		},
301 		stop: function() {
302 			this.commands = null;
303 			this.commands = null;
304 			this.currentCommand = null;
305 			return this.instance.controller.unregisterSequencer(this);
306 		},
307 		getCurrentCommand: function() {
308 			return this.currentCommand;
309 		},
310 		getCommands: function() {
311 			return this.commands;
312 		},
313 		toString: function() {
314 			return "[soma.SequenceCommand]";
315 		}
316

385 	soma.SomaController = soma.extend({
386 		instance:null,
387 		constructor: function(instance) {
388 			this.boundInstance = this.instanceHandler.bind(this);
389 			this.boundDomtree = this.domTreeHandler.bind(this);
390 			this.commands = {};
391 			this.sequencers = {};
392 			this.sequencersInfo = {};
393 			this.lastEvent = null;
394 			this.lastSequencer = null;
395 			this.instance = instance;
396 		},
397 		addInterceptor: function(commandName) {
398 			if (!soma) {
399 				throw new Error("soma package has been overwritten by local variable");
400 			}
401 			if (this.instance.body.addEventListener) {
402 				this.instance.body.addEventListener(commandName, this.boundDomtree, true);
403 			}
404 			this.instance.addEventListener(commandName, this.boundInstance, -Number.MAX_VALUE);
405 		},
406 		removeInterceptor: function(commandName) {
407 			if (this.instance.body.removeEventListener) {
408 				this.instance.body.removeEventListener(commandName, this.boundDomtree, true);
409 			}
410 			this.instance.removeEventListener(commandName, this.boundInstance);
411 		},
412 		executeCommand: function(e) {
413 			var commandName = e.type;
414 			if (this.hasCommand(commandName)) {
415 				var command = new this.commands[commandName]();
416 				command.registerInstance(this.instance);
417 				command.execute(e);
418 			}
419 		},
420 		registerSequencedCommand: function(sequencer, c) {
421 			if (!( c instanceof SequenceCommandProxy )) {
422 				throw new Error("capsulate sequence commands in SequenceCommandProxy objects!");
423 			}
424 			var s = this.sequencersInfo;
425 			if (s[sequencer.id] == null || this.sequencers[sequencer.id] == null) {
426 				this.lastSequencer = sequencer;
427 				s[sequencer.id] = [];
428 				this.sequencers[sequencer.id] = sequencer;
429 			}
430 			c.sequenceId = sequencer.id;
431 			s[sequencer.id].push(c);
432 		},
433 		unregisterSequencedCommand: function(sequencer, commandName) {
434 			if (typeof commandName != "string") {
435 				throw new Error("Controller::unregisterSequencedCommand() expects commandName to be of type String, given:" + commandName);
436 			}
437 			var s = this.sequencersInfo;
438 			if (s[sequencer.id] != null && s[sequencer.id] != undefined) {
439 				var len = s[sequencer.id].length;
440 				for (var i = 0; i < len; i++) {
441 					if (s[sequencer.id][i].event.type == commandName) {
442 						s[sequencer.id][i] = null;
443 						s[sequencer.id].splice(i, 1);
444 						if (s[sequencer.id].length == 0) {
445 							s[sequencer.id] = null;
446 							delete s[sequencer.id];
447 						}
448 						break;
449 					}
450 				}
451 			}
452 		},
453 		unregisterSequencer: function(sequencer) {
454 			var s = this.sequencers;
455 			if (s[sequencer.id] != null && s[sequencer.id] != undefined) {
456 				s[sequencer.id] = null;
457 				delete s[sequencer.id];
458 				s = this.sequencersInfo;
459 				if (s[sequencer.id] != null) {
460 					var len = s[sequencer.id].length;
461 					for (var i = 0; i < len; i++) {
462 						s[sequencer.id][i] = null;
463 					}
464 					s[sequencer.id] = null;
465 					delete s[sequencer.id];
466 					return true;
467 				}
468 			}
469 			return false;
470 		},
471 		hasCommand: function(commandName) {
472 			return this.commands[ commandName ] != null;
473 		},
474 		getCommand: function(commandName) {
475 			if (this.hasCommand(commandName)) {
476 				return this.commands[commandName];
477 			}
478 			return null;
479 		},
480 		getCommands: function() {
481 			var a = [];
482 			var cmds = this.commands;
483 			for (var c in cmds) {
484 				a.push(c);
485 			}
486 			return a;
487 		},
488 		addCommand: function(commandName, command) {
489 			if (this.hasCommand(commandName)) {
490 				throw new Error("Error in " + this + " Command \"" + commandName + "\" already registered.");
491 			}
492 			this.commands[ commandName ] = command;
493 			this.addInterceptor(commandName);
494 		},
495 		removeCommand: function(commandName) {
496 			if (!this.hasCommand(commandName)) {
497 				return;
498 			}
499 			this.commands[commandName] = null;
500 			delete this.commands[commandName];
501 			this.removeInterceptor(commandName);
502 		},
503 		getSequencer: function(event) {
504 			var ss = this.sequencersInfo;
505 			for (var s  in ss) {
506 				var len = ss[s].length;
507 				for (var i = 0; i < len; i++) {
508 					if (ss[s][i] && ss[s][i].event.type === event.type) {
509 						var seq = this.sequencers[ ss[s][i].sequenceId ];
510 						return !!seq ? seq : null;
511 					}
512 				}
513 			}
514 			return null;
515 		},
516 		stopSequencerWithEvent: function(event) {
517 			var ss = this.sequencersInfo;
518 			for (var s in ss) {
519 				var len = ss[s].length;
520 				for (var i = 0; i < len; i++) {
521 					if (ss[s][i].event.type === event.type) {
522 						try {
523 							this.sequencers[ ss[s][i].sequenceId ].stop();
524 						} catch(e) {
525 							return false;
526 						}
527 						return true;
528 					}
529 				}
530 			}
531 			return false;
532 		},
533 		stopSequencer: function(sequencer) {
534 			if (sequencer == null) {
535 				return false;
536 			}
537 			sequencer.stop();
538 			return true;
539 		},
540 		stopAllSequencers: function() {
541 			var ss = this.sequencers;
542 			var sis = this.sequencersInfo;
543 			for (var s in ss) {
544 				if (sis[s] == null) {
545 					continue;
546 				}
547 				var cl = sis[s].length;
548 				sis[s] = null;
549 				delete sis[s];
550 				ss[s].stop();
551 				ss[s] = null;
552 				delete ss[s];
553 			}
554 		},
555 		isPartOfASequence: function(event) {
556 			return ( this.getSequencer(event) != null );
557 		},
558 		getRunningSequencers: function() {
559 			var a = [];
560 			var ss = this.sequencers;
561 			for (var s in ss) {
562 				a.push(ss[s]);
563 			}
564 			return a;
565 		},
566 		getLastSequencer: function() {
567 			return this.lastSequencer;
568 		},
569 		dispose: function() {
570 			for (var nameCommand in this.commands) {
571 				this.removeCommand(nameCommand);
572 			}
573 			for (var nameSequencer in this.sequencers) {
574 				this.sequencers[nameSequencer] = null;
575 				delete this.sequencers[nameSequencer];
576 			}
577 			this.commands = null;
578 			this.sequencers = null;
579 			this.lastEvent = null;
580 			this.lastSequencer = null;
581 		},
582 		domTreeHandler: function(e) {
583 			if (e.bubbles && this.hasCommand(e.type) && !e.isCloned) {
584 				if( e.stopPropagation ) {
585                     e.stopPropagation();
586                 }else{
587                     e.cancelBubble = true;
588                 }
589 				var clonedEvent = e.clone();
590 				// store a reference of the events not to dispatch it twice
591 				// in case it is dispatched from the display list
592 				this.lastEvent = clonedEvent;
593 				this.instance.dispatchEvent(clonedEvent);
594 				if (!clonedEvent.isDefaultPrevented()) {
595 					this.executeCommand(e);
596 				}
597 				this.lastEvent = null;
598 			}
599 		},
600 		instanceHandler: function(e) {
601 			if (e.bubbles && this.hasCommand(e.type)) {
602 				// if the event is equal to the lastEvent, this has already been dispatched for execution
603 				if (this.lastEvent != e) {
604 					if (!e.isDefaultPrevented()) {
605 						this.executeCommand(e);
606 					}
607 				}
608 			}
609 			this.lastEvent = null;
610 		}
611 	});

616 		constructor: function(instance) {
617 			this.views = {};
618 			this.instance = instance;
619 		},
620 		hasView: function(viewName) {
621 			return this.views[ viewName ] != null;
622 		},
623 		addView: function(viewName, view) {
624 			if (this.hasView(viewName)) {
625 				throw new Error("View \"" + viewName + "\" already exists");
626 			}
627 			if (document.attachEvent) {
628 				view.instance = this.instance;

636             }

640 			}

645 				return this.views[ viewName ];
646 			}
647 			return null;
648 		},
649 		getViews: function() {
650 			var clone = {};
651 			for (var name in this.views) {
652 				clone[name] = this.views[name];
653 			}
654 			return clone;
655 		},
656 		removeView: function(viewName) {
657 			if (!this.hasView(viewName)) {
658 				return;
659 			}
660 			if (this.views[viewName]["dispose"] != null) {
661 				this.views[viewName].dispose();
662 			}
663 			this.views[ viewName ] = null;

704 				}

720 			});

723                 events[i].listener.apply((event.srcElement) ? event.srcElement : event.currentTarget, [event]);
