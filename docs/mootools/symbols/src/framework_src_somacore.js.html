168 	var AutoBindProto = {
169 		/** @private */
170 		blackList: ["initialize", "parent", "$constructor", "addEventListener", "removeEventListener" ]
171 		/**
172 		 * AutoBind the application.
173 		 * @name autobind
174 		 * @methodOf soma.AutoBind#
175 		 * @example
176 MyAutoBoundClass = new Class({
177 	Implements: soma.AutoBind,
178 	initialize: function() {
179 		this.autobind();
180 	}
181  });
182 
183 		 */
184 		,autobind: function() {
185 			if (this.wasAutoBound) {
186 				return;
187 			}
188 			var o = this;
189 			var ab = o["AutoBindPattern"];
190 			var coreAb = "([lL]istener|[hH]andler|[cB]allback)$";
191 			if (!ab) {
192 				ab = coreAb;
193 			} else {
194 				ab = coreAb + "|" + ab;
195 			}
196 			for (var k in o) {
197 				if (typeof o[k] == "function") {
198 					if (this._autobindIsBlacklisted(k)) {
199 						continue;
200 					}
201 					if (!k.match(ab)) {
202 						continue;
203 					}
204 					o[k] = o[k].bind(o);
205 				}
206 			}
207 		}
208 		/** @private */
209 		,_autobindIsBlacklisted: function(name) {
210 			var bl = this.blackList;
211 			for (var i = 0; i < bl.length; i++) {
212 				if (bl[i] == name) {
213 					return true;
214 				}
215 			}
216 			return false;
217 		}
